**Step 2: Rewrite `components/login-form.tsx`** — Replace email with Employee ID, wire to login Server Action via `useActionState`, remove OAuth and signup links:

```typescript
"use client"

import { useActionState } from "react"
import { cn } from "@/lib/utils"
import { Button } from "@/components/ui/button"
import {
  Field,
  FieldError,
  FieldGroup,
  FieldLabel,
} from "@/components/ui/field"
import { Input } from "@/components/ui/input"
import { login } from "@/app/actions/auth"
import type { LoginState } from "@/app/actions/auth"

export function LoginForm({
  className,
  ...props
}: React.ComponentProps<"form">) {
  const [state, formAction, isPending] = useActionState<LoginState, FormData>(
    login,
    null,
  )

  return (
    <form
      action={formAction}
      className={cn("flex flex-col gap-6", className)}
      {...props}
    >
      <FieldGroup>
        <div className="flex flex-col items-center gap-1 text-center">
          <h1 className="text-2xl font-semibold">Sign in to your account</h1>
          <p className="text-sm text-balance text-muted-foreground">
            Enter your Employee ID and password
          </p>
        </div>
        {state?.errors?.form && (
          <div role="alert" className="text-sm text-destructive text-center">
            {state.errors.form[0]}
          </div>
        )}
        <Field>
          <FieldLabel htmlFor="employeeId">Employee ID</FieldLabel>
          <Input
            id="employeeId"
            name="employeeId"
            type="text"
            placeholder="Enter your Employee ID"
            required
            autoComplete="username"
            className="bg-background"
          />
          {state?.errors?.employeeId && (
            <FieldError>{state.errors.employeeId[0]}</FieldError>
          )}
        </Field>
        <Field>
          <FieldLabel htmlFor="password">Password</FieldLabel>
          <Input
            id="password"
            name="password"
            type="password"
            required
            autoComplete="current-password"
            className="bg-background"
          />
          {state?.errors?.password && (
            <FieldError>{state.errors.password[0]}</FieldError>
          )}
        </Field>
        <Field>
          <Button type="submit" disabled={isPending} aria-busy={isPending}>
            {isPending ? "Signing in..." : "Sign in"}
          </Button>
        </Field>
      </FieldGroup>
    </form>
  )
}
```

Key changes from template:
- Added `"use client"` (required for `useActionState` hook)
- Replaced email field with Employee ID (type="text", name="employeeId", per D-03)
- Wired form to `login` Server Action via `useActionState`
- Added `isPending` state for button disabled + aria-busy (per UI-SPEC interaction contract)
- Displays `state.errors.form` as a form-level alert for auth failures
- Displays per-field errors from Zod validation
- Removed: "Forgot your password?" link (per D-05: admin-assisted only)
- Removed: "Or continue with" separator + GitHub OAuth button (per UI-SPEC exclusions)
- Removed: "Don't have an account? Sign up" link (per D-01: admin-only creation)
- Heading changed from "Login to your account" to "Sign in to your account" (per UI-SPEC CTA: "Sign in")
- Button text changed from "Login" to "Sign in" (per UI-SPEC copywriting contract)

**Step 2: Create `components/change-password-form.tsx`** — Form with new password, confirm password, and validation:

```typescript
"use client"

import { useActionState } from "react"
import { Button } from "@/components/ui/button"
import {
  Field,
  FieldDescription,
  FieldError,
  FieldGroup,
  FieldLabel,
} from "@/components/ui/field"
import { Input } from "@/components/ui/input"
import { changePassword } from "@/app/actions/auth"
import type { ChangePasswordState } from "@/app/actions/auth"

export function ChangePasswordForm() {
  const [state, formAction, isPending] = useActionState<
    ChangePasswordState,
    FormData
  >(changePassword, null)

  return (
    <form action={formAction} className="flex flex-col gap-4">
      <FieldGroup>
        {state?.errors?.form && (
          <div role="alert" className="text-sm text-destructive text-center">
            {state.errors.form[0]}
          </div>
        )}
        <Field>
          <FieldLabel htmlFor="newPassword">New password</FieldLabel>
          <Input
            id="newPassword"
            name="newPassword"
            type="password"
            required
            autoComplete="new-password"
            className="bg-background"
          />
          <FieldDescription>At least 8 characters.</FieldDescription>
          {state?.errors?.newPassword && (
            <FieldError>{state.errors.newPassword[0]}</FieldError>
          )}
        </Field>
        <Field>
          <FieldLabel htmlFor="confirmPassword">
            Confirm new password
          </FieldLabel>
          <Input
            id="confirmPassword"
            name="confirmPassword"
            type="password"
            required
            autoComplete="new-password"
            className="bg-background"
          />
          {state?.errors?.confirmPassword && (
            <FieldError>{state.errors.confirmPassword[0]}</FieldError>
          )}
        </Field>
        <Field>
          <Button type="submit" disabled={isPending} aria-busy={isPending}>
            {isPending ? "Saving..." : "Set new password"}
          </Button>
        </Field>
      </FieldGroup>
    </form>
  )
}
```

Key design decisions:
- Uses `useActionState` with `changePassword` Server Action from Plan 01
- `FieldDescription` always visible below new-password field: "At least 8 characters." (per UI-SPEC interaction contract)
- Shows per-field errors from Zod validation and password mismatch
- Button label: "Set new password" (per UI-SPEC copywriting contract)
- autoComplete="new-password" on both fields for browser password manager support
- No logout button — proxy.ts enforces this screen, user cannot navigate away
